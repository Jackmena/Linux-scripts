#!/bin/bash

echo "Iniciando sequencia de comandos..."

echo "Criando diretórios..."

mkdir /publico
mkdir /adm
mkdir /ven
mkdir /sec

echo "Diretórios criados com Sucesso!"

echo "Criando grupos..."

groupadd GRP_ADM
groupadd GRP_VEN
groupadd GRP_SEC

echo "Grupos criados com sucesso!"

echo "Adicionando usuários..."

useradd carlos -c "Carlos da Silva" -m -s /bin/bash -p $(openssl passwd -6 Senha123) -G GRP_ADM
passwd carlos -e

useradd maria -c "Maria da Silva" -m -s /bin/bash -p $(openssl passwd -6 Senha123) -G GRP_ADM
passwd maria -e

useradd joao -c "João da Silva" -m -s /bin/bash -p $(openssl passwd -6 Senha123) -G GRP_ADM
passwd joao -e

useradd debora -c "Debora da Silva" -m -s /bin/bash -p $(openssl passwd -6 Senha123) -G GRP_VEN
passwd debora -e

useradd sebastiana -c "Sebastiana da Silva" -m -s /bin/bash -p $(openssl passwd -6 Senha123) -G GRP_VEN
passwd sebastiana -e

useradd roberto -c "Roberto da Silva" -m -s /bin/bash -p $(openssl passwd -6 Senha123) -G GRP_VEN
passwd roberto -e

useradd josefina -c "Josefina da Silva" -m -s /bin/bash -p $(openssl passwd -6 Senha123) -G GRP_SEC
passwd josefina -e

useradd amanda -c "Amanda da Silva" -m -s /bin/bash -p $(openssl passwd -6 Senha123) -G GRP_SEC
passwd amanda -e

useradd rogerio -c "Rogerio da Silva" -m -s /bin/bash -p $(openssl passwd -6 Senha123) -G GRP_SEC
passwd rogerio -e

echo "Usuários adicionados com sucesso!"

echo "Configurando permissões..."

chown root:GRP_ADM /adm
chown root:GRP_VEN /ven
chown root:GRP_SEC /sec

chmod 777 /publico
chmod 770 /adm
chmod 770 /ven
chmod 770 /sec

echo "Configurado permissões com sucesso!"

echo "Finalizado os processos!!!"
